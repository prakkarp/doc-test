    """An interface for the methods that an HTTP Client must implement.

    This class should not be instantiated but should be used as a base class
    for HTTP Client classes.
    """

    def _create_request(
        self, method: str, query_url: str,
        headers: Optional[Mapping[str, Any]] = None,
        query_parameters: Optional[Mapping[str, Any]] = None,
        parameters: Optional[Mapping[str, Any]] = None,
        files: Optional[Mapping[str, Any]] = None
    ) -> http_request.HttpRequest:
        """Common function to create a HttpRequest."""
        if headers is None:
            headers = {}
        if query_parameters is None:
            query_parameters = {}
        if parameters is None:
            parameters = {}
        if files is None:
            files = {}
        return http_request.HttpRequest(
            method, query_url, headers, query_parameters, parameters, files)
