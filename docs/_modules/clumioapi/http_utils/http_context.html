from clumioapi.http_utils import http_request, http_response



[docs]class HttpContext:
    """A Context containing HTTP Request and Response.

    An HTTP Context that contains both the original HttpRequest object that
    intitiated the call and the HttpResponse object that is the result of
    the call.

    Attributes:
        request: The original request object.
        response: The returned response object after executing the request.
            Note that this may be None depending on if and when an error
            occurred.
    """

    def __init__(
        self, request: http_request.HttpRequest, response: http_response.HttpResponse
    ) -> None:
        """Constructor for the HttpContext class.

        Args:
            request: The HTTP Request.
            response: The HTTP Response.
        """
        self.request = request
        self.response = response
