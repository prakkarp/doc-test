from typing import Any, Mapping, Optional

from clumioapi import api_helper



[docs]class HttpRequest:
    """Information about an HTTP Request.

     Includes method, headers, parameters, URL, and Basic Auth details.

    Attributes:
        http_method: The HTTP Method that this request should
            perform when called.
        headers: A dictionary of headers (key : value) that should be
            sent along with the request.
        query_url: The URL that the request should be sent to.
        parameters: A dictionary of parameters that are to be sent along
            with the request in the form body of the request
    """

    def __init__(
        self,
        http_method: str,
        query_url: str,
        headers: Optional[Mapping[str, Any]] = None,
        query_parameters: Optional[Mapping[str, Any]] = None,
        parameters: Optional[Mapping[str, Any]] = None,
        files: Optional[Mapping[str, Any]] = None
    ) -> None:
        """Constructor for the HttpRequest class.

        Args:
            http_method: The HTTP Method.
            query_url: The URL to send the request to.
            headers: The headers for the HTTP Request.
            query_parameters: Query parameters to add in the URL.
            parameters: Form or body parameters to be included in the body.
            files: Files to be sent with the request.
        """
        self.http_method: str = http_method
        self.query_url: str = query_url
        self.headers: Mapping[str, Any] = headers
        self.query_parameters: Mapping[str, Any] = query_parameters
        self.parameters: Mapping[str, Any] = parameters
        self.files: Mapping[str, Any] = files

        """Add a query parameter to the HttpRequest.

            name: The name of the query parameter.
            value: The value of the query parameter.
        """
        self.query_url = api_helper.append_url_with_query_parameters(self.query_url, {name: value})
