from clumioapi.http_utils import http_context



[docs]class APIException(Exception):
    """Class that handles HTTP Exceptions when fetching API Endpoints.

    Attributes:
        response_code: The status code of the response.
        context: The HttpContext of the API call.
    """

    def __init__(self, reason: str, context: http_context.HttpContext) -> None:
        """Constructor for the APIException class.

        Args:
            reason: The reason (or error message) for the Exception
                to be raised.
            context: The HttpContext of the API call.
        """
        super(APIException, self).__init__(reason)
        self.context: http_context.HttpContext = context
        self.response_code: int = context.response.status_code
